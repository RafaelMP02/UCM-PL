int() main() {
	int(int) f(a){};
	bool(int) f(a){};
	f(2); //Error por ser una expresión ambigua
	void () f () {
		return True; //Error, las funciones void no devuelven nada
	};
	int x;
	//En los structs solo se pueden declarar variables (inicializadas o no) y definir otros tipos nuevos
	struct A  {
		int a := 2; //Correcto
		struct B {};
		void() f2() {}; //Error, no se puede declarar una función en un struct
		x := 2; //Error, no se puede realizar una asignación (si no es una declaración) en un struct
	};	
};
